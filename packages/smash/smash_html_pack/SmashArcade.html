        // =============== AUDIO ENGINE (Web Audio API – 100% offline) ===============

        let audioCtx;

        function initAudio() {

            audioCtx = new (window.AudioContext || window.webkitAudioContext)();

           

            // AMBIENT ARCADE HUM (low retro buzz loop)

            let ambientOsc = audioCtx.createOscillator();

            ambientOsc.type = 'sawtooth';

            ambientOsc.frequency.setValueAtTime(48, audioCtx.currentTime);

            let ambientGain = audioCtx.createGain();

            ambientGain.gain.value = 0.08;

            let filter = audioCtx.createBiquadFilter();

            filter.type = 'lowpass'; filter.frequency.value = 800;

            ambientOsc.connect(filter).connect(ambientGain).connect(audioCtx.destination);

            ambientOsc.start();

           

            // 8-BIT SOUNDTRACK LOOP (einfache Chiptune-Melodie)

            let melody = [262, 330, 392, 523, 392, 330, 262, 392];

            let noteIndex = 0;

            setInterval(() => {

                let osc = audioCtx.createOscillator();

                osc.type = 'square';

                osc.frequency.value = melody[noteIndex];

                let gain = audioCtx.createGain();

                gain.gain.value = 0.15;

                osc.connect(gain).connect(audioCtx.destination);

                osc.start();

                gain.gain.setValueAtTime(0.15, audioCtx.currentTime);

                gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.25);

                osc.stop(audioCtx.currentTime + 0.3);

                noteIndex = (noteIndex + 1) % melody.length;

            }, 280);

        }

        function playCoin() {

            let osc = audioCtx.createOscillator();

            osc.type = 'sine';

            osc.frequency.setValueAtTime(800, audioCtx.currentTime);

            let gain = audioCtx.createGain();

            osc.connect(gain).connect(audioCtx.destination);

            osc.start();

            gain.gain.setValueAtTime(0.6, audioCtx.currentTime);

            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.8);

            osc.frequency.exponentialRampToValueAtTime(200, audioCtx.currentTime + 0.8);

            osc.stop(audioCtx.currentTime + 0.9);

        }

        function playBeep(freq, duration, type='square', vol=0.3) {

            let osc = audioCtx.createOscillator();

            osc.type = type;

            osc.frequency.value = freq;

            let gain = audioCtx.createGain();

            gain.gain.value = vol;

            osc.connect(gain).connect(audioCtx.destination);

            osc.start();

            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);

            osc.stop(audioCtx.currentTime + duration);

        }

        function onKey(e){ /* WASD wie vorher */ if(e.type==='keydown'){ if(e.code==='KeyW') move.f=true; if(e.code==='KeyS') move.b=true; if(e.code==='KeyA') move.l=true; if(e.code==='KeyD') move.r=true; } else { if(e.code==='KeyW') move.f=false; if(e.code==='KeyS') move.b=false; if(e.code==='KeyA') move.l=false; if(e.code==='KeyD') move.r=false; } }

            if(hits.length){

                playBeep(1200, 0.1);

                }

            }

        }

        function animate(){
